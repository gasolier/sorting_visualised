function bubbleSort(arr) {
    let n = arr.length;
    while (true) {
        let swapped = false;
        for (let i = 0; i < n; i++) {
            if (arr[i] > arr[i + 1]) {
                arr = swap(i, i + 1, arr);
                swapped = true;
            }
        }
        n--;
        if (!swapped) {
            break;
        }
    }
    return arr;
}
